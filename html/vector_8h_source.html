31 /**
32  * @brief A generic and dynamic container that expands
33  * automatically as elements are added.
34 
35  * The library supports amortized constant time
36  * insertion and removal of elements , as well as
37  * constant time access.
38  *
39  * Memory leaks free. If one function in the library
40  * creates a pointer to Heap, then the library supplies
41  * a function to free it.
42  *
43  * The restriction in C to use generic container is
44  * based on this language does not allow data types
45  * to be passed as parameters. That means a generic
46  * container needs to manually manage memory in
47  * terms of the client element size, not client
48  * data type.
49  */
50 typedef struct Vector{
51  int len; /**< Number of elements in Vector object*/
52 
53  void *data; /**< Array to hold the information */
54 
55  int capacity; /**< Capacity, in number of elements. Capacity has
56  to be greater than Zero*/
57 
58  size_t ele_size; /**< Vector's data type.
59  It has to be the returned value of 'sizeof(Vector's
60  data type)'
61 
62  The constructor requires that the client element
63  size be identified up front.
64 
65  Since it doesn’t (and will never) know the true data
66  type, it needs the size so it at least knows how
67  many bytes to replicate behind the scenes with
68  every call to manage data.*/
69 
70  int (*compar)(const void *, const void *); /**<
71  Comparing Function pointer*/
72  void(*Destroy)(struct Vector *v);/**<Destructor*/
73  void(*Destroy_pointer)(struct Vector *v);/**<Destructor to use
74  with the function vector_Insert_from_file. It frees
75  the pointers created and then it calls Destroy**/
76 
77 } vector_t;
