295 /**
296  * @brief Binary Search of an item. It copies in 'vector_t *slice'
297  * all the items that match with the supplied value by
298  * the client.
299  * @param v A pointer to vector_t objet
300  * @param value Item searched
301  * @param slice Pointer to vector_t object which will collect all the
302  * items that match with the supplied value by the
303  * client.
304  * @returns V_OK if the element was found, or
305  * V_IS_EMPTY if the Vector Object is empty or
306  * V_ERR_VALUE_NOT_FOUND.
307  */
308 v_stat vector_Filter(vector_t *v, void *value, vector_t *slice) {
309 
310  if (vector_isEmpty(v))
311  return V_ERR_IS_EMPTY;
312  else {
313  vector_Sort(v);
314  int left = 0;
315  int right = v->len - 1;
316 
317  while (left <= right) {
318  int middle = (left + right) / 2;
319  void *item = (char *)v->data + v->ele_size * middle;
320  int compar = v->compar( item, value);
321 
322  if (compar == 0) {
323  int reverse = middle;
324  do {
325  vector_Insert(slice, item);
326  middle++;
327  item = (char *)v->data + v->ele_size * middle;
328  compar = v->compar(value, item);
329  } while (compar == 0 && (middle < (v->len - 1)));
330 
331  if (reverse > 0) {
332  middle = reverse - 1;
333  item = (char *)v->data + v->ele_size * middle;
334  compar = v->compar(value, item);
335 
336  while (compar == 0 && (middle > 0)) {
337  vector_Insert(slice, item);
338  middle--;
339  item = (char *)v->data + v->ele_size * middle;
340  compar = v->compar(value, item);
341  }
342  }
343  vector_Sort(slice);
344  return V_OK;
345 
346  } else if (compar > 0)
347  right = middle - 1;
348 
349  else
350  left = middle + 1;
351  }
352  }
353  return V_ERR_VALUE_NOT_FOUND;
354 }

396  * @brief Returns the strings that matches the pattern in one slice.
397  * In this case, the search is, at least, Big-O lineal time of
398  * complexity.
399  * We use the C's library 'fnmatch' to pattern matching.
400  * The pattern may include the following special characters:
401  *
402  * * Matches zero of more characters.
403  * ? Matches exactly one character.
404 
405  * [...] Matches one character if it's in a range of characters.
406  * If the first character is `!', matches if the character is
407  * not in the range. Between the brackets, the range is
408  * specified by listing the characters that are in the range,
409  * or two characters separated by `-' to indicate all
410  * characters in that range.
411  * For example, `[a-d]' matches `a', `b', `c', or `d'.
412  * If you want to include the literal `-' in the range, make
413  * it the first character, like in `[-afz]'.
414 
415  * \ Causes the next character to not be treated as a
416  * wildcard. For example, `\*' matches an asterisk.
417  *
418  * @param v A pointer to vector_t objet
419  * @param pattern Pattern searched. It is a string.
420  * @param slice Vector object pointer to return the slice
421  * @returns V_OK if the element was found, or
422  * V_IS_EMPTY if the Vector Object is empty or
423  * V_ERR_VALUE_NOT_FOUND.
424  */
425 v_stat vector_Pattern(vector_t *v, const char *pattern,
426  vector_t *slice) {
427 
428  if (vector_isEmpty(v))
429  return V_ERR_IS_EMPTY;
430  else {
431  size_t index;
432  const void *item;
433 
434  for (index = 0; index < v->len; index++) {
435  item = (char *)v->data + v->ele_size * index;
436  if (!fnmatch(pattern, item, 0))
437  vector_Insert(slice, item);

439  vector_Sort(slice);

528  * @param item Value to insert in Vector object
529  */
530 void vector_Insert(vector_t*v, void const *item) {
531 
532  void *destAddr;
533  if (v->len == v->capacity) {
534  v->capacity *= 2; // Capacity is duplicated
535  v->data = realloc(v->data, v->capacity * v->ele_size);
536  // If memory error, exit,,,
537  assert(v->data != NULL);
538  }
539 
540  /* v->len is Base One Index.
541  * First we calculate his direction in memory; then we copy

544  destAddr = (char *)v->data + v->ele_size * v->len;
545  assert(memcpy(destAddr, item, v->ele_size));
546  v->len++;
547 }
